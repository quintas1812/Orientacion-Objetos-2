interface A {
void metodo();
}
interface B {
void metodo(String b);
}
interface C {
boolean metodo(String c);
}
interface D<T, R> {
 R metodo(T c);
}
public class AprendiendoLambdas {
public void unMetodo(A a) {
a.metodo();
}
public void unMetodo(B b) {
b.metodo("unString");
}
public void unMetodo(C c) {
System.out.println(c.metodo("otroString") ? "true" : "false");
}
public void unMetodo(D<Long, Long> d) {
d.metodo(10L);
}
}

public class Main {
public static void main(String[] args) {
AprendiendoLambdas a = new AprendiendoLambdas();                        // RESPUESTAS
a.unMetodo((b) → { System.out.println("abcd" + b);}); 			        // este metodo invoca unMetodo(B b)
a.unMetodo(() → System.out.println("abcd"));				            // este metodo invoca unMetodo(A a)
a.unMetodo((variable) → {System.out.println("abcd");});			        // este metodo invoca unMetodo(B b)
a.unMetodo((variable) → {System.out.println("abcd"); return true;});    // este metodo invoca unMetodo(C c)
a.unMetodo((Long variable) -> {						                    // este metodo invoca unMetodo(D<Long, Long> d)
System.out.println("abcd");
return 10L;
});
}
}
